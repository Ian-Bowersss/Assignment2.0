 function ErrorRegister() {
    for (let key in errors) {
        const inputField = document.querySelector(`[name="${key}"]`); // Use querySelector instead of getElementById
        if (inputField) {
            const errorSpan = document.createElement('span');
            errorSpan.textContent = errors[key];
            errorSpan.style.color = 'red';
            errorSpan.style.fontWeight = 'bold'; // Make error messages bold
            if (inputField.nextSibling) {
                inputField.parentNode.insertBefore(errorSpan, inputField.nextSibling);
            } else {
                inputField.parentNode.appendChild(errorSpan);
            }
        } else {
            console.error('Input field with name', key, 'not found.'); // Log error
        }
    }
}


// Call displayErrors when the window loads
window.onload = function () {
    ErrorRegister();
};
